# 가상환경

1. Anaconda Prompt 관리자로 실행
2. conda install scikit-learn
3. conda  --version
    1. 버전확인
4. conda info
    1. 설치정보 확인
5. conda create  -name 가상환경명 python=3.5
    1. conda create  -n 가상환경명 python=3.5
    2. 가상환경 만들기 파이썬 3.5 버전
6. conda remove  --name 가상환경명 --all
    1. 가상환경 지우기
7. conda activate test
    1. 가상환경으로 변경
8. conda deactivate
